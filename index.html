◈ GET STARTED

Up and running
in 5 minutes.

# Install
pip install greencode

# Measure your project
greencode measure

# Custom command
greencode measure --command "pytest tests/ -x"

# View history
greencode history --limit 20

# Generate badge manually
greencode badge --owner myname --repo mylib
# .github/workflows/carbon.yml

name: Carbon Footprint
on: [push, pull_request]

jobs:
measure:
runs-on: ubuntu-latest
steps:
- uses: actions/checkout@v4
- uses: AlexAp0st0l/GreenCode@v1
with:
country: DEU
comment-on-pr: true
from greencode import measure, generate

# Measure CO₂ while running tests
result = measure(path=".", country_iso="DEU")

print(result["grams_co2"]) # 4.2100
print(result["grade"]) # "green"

# Generate SVG badge
svg = generate(result["grams_co2"], result["grade"])
Path(".greencode/badge.svg").write_text(svg)
